# Sort items by frequency count
def freq_sort(items):
  counts = {}
  for x in items:
    counts[x] = counts.get(x, 0) + 1
  return sorted(items,
    key=lambda x: -counts[x])

result = freq_sort([
  "cat", "dog", "cat",
  "bird", "cat", "dog"
])
print(result)
